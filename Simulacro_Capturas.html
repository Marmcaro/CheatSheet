# Calcular el primer y tercer cuartil (Q1 y Q3)
Q1 <- quantile(datos$Peso, 0.25)
Q3 <- quantile(datos$Peso, 0.75)

# Calcular el rango intercuartÃ­lico (IQR)
IQR <- Q3 - Q1

# Calcular los lÃ­mites para los valores atÃ­picos

# Donde Q1 es el primer cuartil (25 percentil),
